    public void Loadfacultyid()
    {
        Connection();
        try {
        pst = con.prepareStatement("select * from leaveinfo");
        ResultSet rs= pst.executeQuery();
        
        while(rs.next())
        {
            txtid.addItem(rs.getString("facultyid"));
        
        }
        } catch(SQLException ex){
            Logger.getLogger(Applyleave.class.getName()).log(Level.SEVERE, null, ex);
        } 
        
    } 
    
    public void Loadleave()
    {
        Connection();
     
        String facultyid =(String)txtid.getSelectedItem();
        
        try {
            pst = con.prepareStatement("select * from leavecal where facultyid=?");
            pst.setString(1,facultyid);
            ResultSet rs2=pst.executeQuery();
        
            if(rs2.next() == false)
            {
                JOptionPane.showMessageDialog(this,"Leave information error");
            }
            else
            {
              casual = rs2.getInt("casual"); 
              annual = rs2.getInt("annual");
              medical = rs2.getInt("medical");
            }
            
            
        } catch(SQLException ex) {
            Logger.getLogger(Applyleave.class.getName()).log(Level.SEVERE, null, ex);
        }
    
    }

    private void txtidActionPerformed(java.awt.event.ActionEvent evt) {                                      
        // TODO add your handling code here:
        String facultyid =  txtid.getSelectedItem().toString();
        
        Connection();
        try {
            pst = con.prepareStatement("select * from leaveinfo where facultyid = ?");
            pst.setInt(1,Integer.parseInt( facultyid));
            ResultSet rs1=pst.executeQuery();
            
            if(rs1.next() == false)
            {
                JOptionPane.showMessageDialog(this,"Faculty id error");
            }
            else
            {
                String name = rs1.getString("name");
                txtname.setText(name.trim());
                
            }
        } catch (SQLException ex){
            Logger.getLogger(Applyleave.class.getName()).log(Level.SEVERE, null, ex);
        }
    }                                     

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        Loadleave();
        int bleave ;
        String facultyid = (String) txtid.getSelectedItem();
         
         int days = Integer.parseInt(txtdays.getValue().toString());
         
         if(rcasual.isSelected() == true )
         {
             bleave = casual-days;
             
           if(bleave < 0)
           {
               JOptionPane.showMessageDialog(this,"You have a casual leave" + casual);
           }
           else
           {
             try {
                 pst = con.prepareStatement("update leavecal set casual=? where facultyid=?");
                 pst.setInt(1,bleave);
                 pst.setString(2,facultyid);
                 pst.executeUpdate();
                 
                 JOptionPane.showMessageDialog(this,"your casual leave updated");
                 
             }   catch (SQLException ex) {  
                     Logger.getLogger(Applyleave.class.getName()).log(Level.SEVERE, null, ex);
                 }  
           }
         }
         
         else if(rannual.isSelected() == true )
         {
            bleave = annual-days;
             
           if(bleave < 0)
           {
               JOptionPane.showMessageDialog(this,"You have a annual leave" + annual);
           }
           else
           {
             try {
                 pst1 = con.prepareStatement("update leavecal set annual=? where facultyid=?");
                 pst1.setInt(1,bleave);
                 pst1.setString(2,facultyid);
                 pst1.executeUpdate();
                 
                 JOptionPane.showMessageDialog(this,"your annual leave updated");
                 
             }   catch (SQLException ex) {  
                     Logger.getLogger(Applyleave.class.getName()).log(Level.SEVERE, null, ex);
                 }  
           }
             
         }
         else if(rmedical.isSelected() == true )
         {
            bleave = medical-days;
             
           if(bleave < 0)
           {
               JOptionPane.showMessageDialog(this,"You have a medical leave" + medical);
           }
           else
           {
             try {
                 pst2 = con.prepareStatement("update leavecal set medical = ? where facultyid = ?");
                 pst2.setInt(1,bleave);
                 pst2.setString(2,facultyid);
                 pst2.executeUpdate();
                 
                 JOptionPane.showMessageDialog(this,"your medical leave updated");
                 
             }   catch (SQLException ex) {  
                     Logger.getLogger(Applyleave.class.getName()).log(Level.SEVERE, null, ex);
                 }  
           }
             
         }
         else
         {
            JOptionPane.showMessageDialog(this,"something wrong"); 
         }
    }
