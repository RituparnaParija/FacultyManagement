    public void LoadEmpNo(){
      
       
        try{
            pst = con.prepareStatement("select empid from facultyprofile");
            rs = pst.executeQuery();
            txtid.removeAllItems();
            while(rs.next()){
                txtid.addItem(rs.getString("empid"));
                
            }
            
        } catch (SQLException ex) {
            Logger.getLogger(FrmMainProfile.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        String empid =  txtid.getSelectedItem().toString();
        try{
            pst = con.prepareStatement("select * from facultyprofile where empid = ?");
            pst.setString(1, empid);
            rs = pst.executeQuery();
            
            if(rs.next()==true){
                txtname.setText(rs.getString("name"));
                txtage.setText(rs.getString("age"));
                txtgender.setText(rs.getString("gender"));
                txtcontact.setText(rs.getString("contact"));
                txtemail.setText(rs.getString("email"));
                txtaddress.setText(rs.getString("address"));
                txtdepartment.setText(rs.getString("department"));
                txtjoiningdate.setText(rs.getString("joiningdate"));
                txtdesignation.setText(rs.getString("designation"));
            }
        } catch (SQLException ex) {
            Logger.getLogger(FrmMainProfile.class.getName()).log(Level.SEVERE, null, ex);
        }
        
    }
